# Tuple on one line

{foo, bar, baz}

==>

Query(Statements(PipelineStatement(Pipeline(TupleExpression(Ident,Ident,Ident)))))

# Tuple on multiple lines

{
  foo,
  bar,
  baz
}

==>

Query(Statements(PipelineStatement(Pipeline(TupleExpression(Ident,Ident,Ident)))))

# Tuple on multiple lines with blank lines

{

  foo,

  bar,

  baz

}

==>

Query(Statements(PipelineStatement(Pipeline(TupleExpression(Ident,Ident,Ident)))))

# Tuple with key and value

{foo=bar}

==>

Query(Statements(PipelineStatement(Pipeline(TupleExpression(AssignCall(Ident))))))
